// login.js — logowanie z użyciem localStorage document.getElementById("loginForm").addEventListener("submit", function (e) { e.preventDefault(); const uid = document.getElementById("uid").value; const password = document.getElementById("password").value; // Pobierz użytkowników z localStorage const users = JSON.parse(localStorage.getItem("users")) || {}; if (users[uid] && users[uid].password === password) { // Zapisz dane do sesji (localStorage) localStorage.setItem("uid", uid); localStorage.setItem("role", users[uid].role); // Przekieruj do dashboard window.location.href = "dashboard.html"; } else { alert("Nieprawidłowy UID lub hasło."); } });
